Public Class Form1

    Friend WithEvents ScrSelector As New ScreenRegionSelector With
        {
            .BorderColor = Color.GreenYellow,
            .BorderSize = 2,
            .BackgroundColor = Color.YellowGreen,
            .BackgroundOpacity = 0.04R
        }

    Private Sub Button_Click() Handles Button1.Click

        Me.ScrSelector.Show()

    End Sub

    Private Sub ScrSelector_RegionSelected(ByVal sender As Object, ByVal e As RegionSelectedEventArgs) _
    Handles ScrSelector.RegionSelected

        Dim sb As New Global.System.Text.StringBuilder
        With sb
            .AppendLine("Selected Area")
            .AppendLine()
            .AppendLine("[+] Size")
            .AppendLine(String.Format("Width: {0}", CStr(e.Region.Width)))
            .AppendLine(String.Format("Height: {0}", CStr(e.Region.Height)))
            .AppendLine()
            .AppendLine("[+] Coordinates")
            .AppendLine(String.Format("Top: {0}", CStr(e.Region.Top)))
            .AppendLine(String.Format("Left: {0}", CStr(e.Region.Left)))
            .AppendLine(String.Format("Right: {0}", CStr(e.Region.Right)))
            .AppendLine(String.Format("Bottom: {0}", CStr(e.Region.Bottom)))
            .AppendLine()
        End With

        MessageBox.Show(sb.ToString())

    End Sub

End Class
