    Friend FormLock As New WindowLocker(Me)

    Private Sub Form1_Load() Handles MyBase.Shown

        With Me.FormLock
            .CloseButtonEnabled = False
            .MaximizeButtonEnabled = False
            .MinimizeButtonEnabled = False
            .MoveEnabled = False
            .MenuEnabled = False
        End With

    End Sub
