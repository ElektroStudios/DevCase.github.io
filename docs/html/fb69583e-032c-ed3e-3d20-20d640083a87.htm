VB
Dim loginAttempt As New WebsiteLoginRequest With {
    .Address = New Uri("http://www.planetsuzy.org/login.php?do=login"),
    .Credential = New NetworkCredential("username", "password"),
    .SuccessConfirmationString = $"Thank you for logging in, { .Credential.UserName}.",
    .RequestData = New NameValueCollection From {
                        {"do", "login"},
                        {"cookieuser", "1"},
                        {"vb_login_username", .Credential.UserName},
                        {"vb_login_password", .Credential.Password}
    }
}

Dim loginResult As WebsiteLoginResponse = loginAttempt.GetLogInResponse()

Console.WriteLine($"Log-In Address..: {loginAttempt.Address}")
Console.WriteLine($"Log-In Req. Data: {loginAttempt.RequestData.ToQueryString()}")
Console.WriteLine($"Log-In Confirm..: {loginAttempt.SuccessConfirmationString}")
Console.WriteLine($"Log-In Success?.: {loginResult.Success}")
Console.WriteLine()
Console.WriteLine("Response Cookies:")
For Each cookie As Cookie In loginResult.ResponseCookies
    Console.WriteLine("{0}: {1}", cookie.Name, cookie.Value)
