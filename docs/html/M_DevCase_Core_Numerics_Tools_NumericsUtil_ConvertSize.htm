Dim byteToByte As Double = NumericsUtil.Convertsize(92928374, SizeUnits.Byte, SizeUnits.Byte)

Console.WriteLine(String.Format("92928374 bytes = {0} MB", byteToMb.ToString("n2"))) ' Result: 88,62 MB
